<script>
  async function startCollectionProcess() {
    const btn = document.getElementById('startButton');
    const out = document.getElementById('results');

    btn.disabled = true;
    out.innerHTML = "Running diagnostics... Please wait.";

    try {
      // Ensure the wrapper is loaded
      if (!window.fpDiagnostics || typeof window.fpDiagnostics.collectDiagnosticsAndSubmit !== 'function') {
        out.innerHTML = "❌ <strong>CRITICAL JS ERROR:</strong> Wrapper script not loaded. Check the /static/fp_wrapper.js path and console.";
        btn.disabled = false;
        return;
      }

      // Call the SAFE diagnostics collector (non-spoofing)
      const finalStatus = await window.fpDiagnostics.collectDiagnosticsAndSubmit({
        testMode: false,               // set true to generate synthetic samples
        endpoint: '/api/submit-fingerprints',
        includeCanvasMock: false       // set true if you want a placeholder canvas hash
      });

      if (finalStatus && finalStatus.status === 'success') {
        out.innerHTML = "✅ <strong>SUBMISSION COMPLETE!</strong> Thank you for your contribution. You may close this window.";
      } else {
        out.innerHTML = "❌ <strong>ERROR:</strong> Submission failed. Please try again.";
        btn.disabled = false;
      }
    } catch (e) {
      console.error("Uncaught Error in Start Process:", e);
      out.innerHTML = "❌ <strong>CRITICAL JS ERROR:</strong> Check console for details.";
      btn.disabled = false;
    }
  }

  // Expose globally for the onclick handler
  window.startCollectionProcess = startCollectionProcess;
</script>
